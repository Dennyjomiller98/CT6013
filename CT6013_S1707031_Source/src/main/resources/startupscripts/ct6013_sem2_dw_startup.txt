DROP TABLE DW_AUTH CASCADE CONSTRAINTS;
CREATE TABLE DW_AUTH
(
  USERNAME VARCHAR2(100) NOT NULL
, PIN VARCHAR2(4) NOT NULL
, PIN_START VARCHAR2(100) NOT NULL
, PIN_END VARCHAR2(100) NOT NULL
);

DROP TABLE DW_USERS CASCADE CONSTRAINTS;
CREATE TABLE DW_USERS
(
  USERNAME VARCHAR2(100) NOT NULL
, PWORD VARCHAR2(255) NOT NULL
, ROLE VARCHAR2(50) NOT NULL
);
